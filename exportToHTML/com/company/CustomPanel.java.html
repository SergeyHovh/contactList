package com.company; 
 
import javax.swing.*; 
import java.awt.event.ActionEvent; 
import java.util.Vector; 
 
public class CustomPanel extends JPanel implements ButtonActivity { 
 
    private JButton personalInfo = new JButton(); 
    private JButton delete = new JButton("Delete"); 
    private Vector<CustomPanel> tempVector = Frame.getCustomPanelVector(); 
    CustomPanel() { 
        setSize(Frame.WIDTH, 80); 
        personalInfo.setBounds(0, 0, ((int) (super.getWidth() * 0.75)), super.getHeight()); 
        delete.setBounds(personalInfo.getWidth(), 0, ((int) (super.getWidth() * 0.25)), super.getHeight()); 
        add(personalInfo); 
        add(delete); 
        delete.addActionListener(new AbstractAction() { 
            @Override 
            public void actionPerformed(ActionEvent e) { 
                Delete(); 
            } 
        }); 
    } 
 
    public JButton getDelete() { 
        return delete; 
    } 
 
    public JButton getPersonalInfo() { 
        return personalInfo; 
    } 
 
    @Override 
    public void Delete() { 
        super.remove(personalInfo); 
        super.remove(delete); 
        tempVector.remove(this); 
        Main.updatePanel(Frame.getUpperPanel(), tempVector); 
    } 
}

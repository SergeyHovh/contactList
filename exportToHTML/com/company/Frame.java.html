package com.company; 
 
import javax.swing.*; 
import java.awt.event.ActionEvent; 
import java.util.Vector; 
 
public class Frame extends JFrame { 
    private static JPanel upperPanel = new JPanel(null); 
    private static JPanel lowerPanel = new JPanel(null); 
    private static JButton b = new JButton("Add"); 
    final static public int WIDTH = 800; 
    final static public int HEIGHT = 600; 
    private static Vector<CustomPanel> customPanelVector = new Vector<CustomPanel>(10,10); 
 
    Frame() { 
        setSize(WIDTH,HEIGHT); 
        upperPanel.setBounds(0, 0, super.getWidth(), 3 * super.getHeight() / 4); 
        lowerPanel.setBounds(0, upperPanel.getHeight(), super.getWidth(), super.getHeight() / 4); 
        setResizable(false); 
        lowerPanel.add(b); 
        b.setSize(lowerPanel.getSize()); 
        b.addActionListener(new AbstractAction() { 
            @Override 
            public void actionPerformed(ActionEvent e) { 
                addNewButton(upperPanel); 
            } 
        }); 
        super.add(upperPanel); 
        super.add(lowerPanel); 
//        upperPanel.add(upperScrollPanel); 
//        upperScrollPanel.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS); 
        setLayout(null); 
        setVisible(true); 
        exit(); 
    } 
 
    private void addNewButton(JComponent panel) { 
        CustomPanel c = new CustomPanel(); 
        customPanelVector.add(c); 
        Main.updatePanel(upperPanel, customPanelVector); 
    } 
 
    private void exit() { 
        setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE); 
    } 
 
    public static JPanel getUpperPanel() { 
        return upperPanel; 
    } 
 
    public static Vector<CustomPanel> getCustomPanelVector() { 
        return customPanelVector; 
    } 
}

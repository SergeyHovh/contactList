package com.company; 
 
import javax.swing.*; 
import java.util.Vector; 
 
public class Main { 
 
    public static void main(String[] args) { 
        Frame f = new Frame(); 
    } 
 
    public static void updatePanel(JComponent panel, Vector<CustomPanel> customPanelVector) { 
        panel.removeAll(); 
        int buttonID = 0; 
        for(CustomPanel current: customPanelVector) { 
            current.setSize(panel.getWidth(), current.getHeight()); 
            current.setLocation(0, current.getHeight() * buttonID); 
            buttonID++; 
            current.getPersonalInfo().setName(buttonID + ""); 
            panel.add(current); 
        } 
        panel.repaint(); 
    } 
}
